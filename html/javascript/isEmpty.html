/** * d 数据 * t o == Object、a == Array、N == Number、s == String、u == undefined、n == null、b==Boolean */ const dataType = (d, t) => { if(t){ let i = Object.prototype.toString.call(d) let f = true; if(t == "o")i == "[object Object]" ? f=true : f=false; if(t == "s")i == "[object String]" ? f=true : f=false; if(t == "N")i == "[object Number]" ? f=true : f=false; if(t == "n")i == "[object null]" ? f=true : f=false; if(t == "u")i == "[object undefined]" ? f=true : f=false; if(t == "b")i == "[object Boolean]" ? f=true : f=false; return f; }else{ return Object.prototype.toString.call(d) } }
