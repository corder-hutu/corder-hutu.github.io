js代码

    window.onload = function(){
        //获取div box
        var box = document.getElementById("box");
        //红色div跟随鼠标移动
        //为document绑定跟随鼠标移动事件

        box.style.left = 800 + "px";
        box.style.top = 0;

        document.onmousemove = function(ev){
            //处理event对象兼容性
            ev = ev || event;

            //获取鼠标指针的坐标
            //clientX和clientY获取到的是鼠标相对于当前可视窗口的坐标
            var x = ev.clientX;
            var y = ev.clientY;

            //获取滚动条的高度
            //document.body  只支持chrome浏览器
            //document.documentElement  支持火狐 ie
            var st = document.body.scrollTop || document.documentElement.scrollTop;
            var sl = document.body.scrollLeft || document.documentElement.scrollLeft;

            /**
            * pageX和pageY表示鼠标相对于整个页面的坐标
            * 但是IE8以下的浏览器，不支持这个属性
            */
            /*var x = ev.pageX;
            var y = ev.pageY;*/

            //修改div的top和left
            box.style.top = y + st + "px";
            box.style.left = x + sl - 201 + "px";
        }
    }
